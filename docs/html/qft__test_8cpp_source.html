29 void qft(QubitRegister<Type> &psi)
30 {
31  int n = openqu::ilog2(psi.size());
32 
33  // main computation
34  for (int i = n - 1; i >= 0; i--)
35  {
36  for (int j = n - 1; j > i; j--)
37  {
38  int k = j - i;
39  openqu::TinyMatrix<Type, 2, 2, 32> phaseshift;
40  phaseshift(0, 0) = {1, 0};
41  phaseshift(0, 1) = {0, 0};
42  phaseshift(1, 0) = {0, 0};
43  phaseshift(1, 1) = Type(cos(M_PI / D(UL(1) << UL(k))), sin(M_PI / D(UL(1) << UL(k))));
44  psi.ApplyControlled1QubitGate(j, i, phaseshift);
45  }
46  psi.ApplyHadamard(i);
47  }
48 
49  // perform swapping
50  for (int i = 0; i < (n / 2); i++) {
51  psi.ApplySwap(i, n - 1 - i);
52  }
53 }
