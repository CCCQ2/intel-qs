40  // Steps: 1. 2. 3. 4.
41  // i j | i j | i j | i j
42  // s1 d1 | s1 d1 | s1 d1&s1 | s1&d1 d1&s1
43  // s2 d2 | s2 d2 | s2&d2 d2 | s2&d2 d2&s2
44  // T T | d2 s1 | d2&s2 s1&d1 | T T

48  std::size_t glb_start = UL(myrank) * localSize();

50  // std::string s;

53  {
54  itask = myrank;
55  jtask = itask + (1 << pstrideexp);

106  }

177 bool QubitRegister<Type>::Apply1QubitGate_helper(unsigned qubit_, TM2x2<Type> const&m,
178  std::size_t sind, std::size_t eind)
179 {
180  assert(qubit_ < num_qubits);
181  unsigned qubit = (*permutation)[qubit_];
182  assert(qubit < num_qubits);
183 
184  TODO(Add diagonal special case)
185 
186  unsigned myrank=0, nprocs=1, log2_nprocs=0;
187 #ifdef INTELQS_HAS_MPI
188  myrank = openqu::mpi::Environment::rank();
189  nprocs = openqu::mpi::Environment::size();
190  log2_nprocs = openqu::ilog2(openqu::mpi::Environment::size());
191 #endif
192  unsigned M = num_qubits - log2_nprocs;
193  std::size_t P = qubit;
194 
195  std::size_t src_glb_start = UL(myrank) * localSize();
196  // check for special case of diagonal
197  bool diagonal = (m[0][1].real() == 0. && m[0][1].imag() == 0. &&
198  m[1][0].real() == 0. && m[1][0].imag() == 0.);
199 
200  std::string gate_name = "SQG("+openqu::toString(P)+")::"+m.name;
201 
202  if (timer)
203  timer->Start(gate_name, P);
204 
205  if (P < M)
206  {
207  assert(eind - sind <= localSize());
208  Loop_DN(sind, eind, UL(P), state, state, 0UL, (1UL << P), m, specialize, timer);
209  }
210  else
211  {
212  assert(eind - sind == localSize());
213  if (specialize && diagonal)
214  {
215  if (check_bit(src_glb_start, P) == 0 )
216  ScaleState(sind, eind, state, m[0][0], timer);
217  else
218  ScaleState(sind, eind, state, m[1][1], timer);
219  }
220  else
221  {
222  HP_Distrpair(P, m);
223  }
224  }
225 
226  if (timer)
227  timer->Stop();
228 
229  return true;
230 }
