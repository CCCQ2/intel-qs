190 bool QubitRegister<Type>::GetClassicalValue(unsigned qubit, BaseType tolerance) const
191 {
192  unsigned myrank=0, nprocs=1, log2_nprocs=0;
193 #ifdef INTELQS_HAS_MPI
194  myrank = openqu::mpi::Environment::rank();
195  nprocs = openqu::mpi::Environment::size();
196  log2_nprocs = openqu::ilog2(nprocs);
197 #endif
198  unsigned M = num_qubits - log2_nprocs;
199 
200  std::size_t delta = 1UL << qubit;
201  int bit_is_zero = 0, bit_is_one = 0;
202  if (qubit < M)
203  {
204  for (std::size_t i = 0; i < localSize(); i += 2 * delta)
205  {
206  for (std::size_t j = 0; j < delta; ++j)
207  {
208  if (std::norm(state[i + j]) > tolerance)
209  {
210  bit_is_zero = 1;
211  goto done;
212  }
213  if (std::norm(state[i + j + delta]) > tolerance)
214  {
215  bit_is_one = 1;
216  goto done;
217  }
218  }
219  }
220  }
221  else
222  {
223  std::size_t src_glb_start = UL(myrank) * localSize();
224  if (check_bit(src_glb_start, qubit) == 0)
225  {
226  for (std::size_t j = 0; j < localSize(); j++)
227  {
228  if (std::norm(state[j]) > tolerance)
229  {
230  bit_is_zero = 1;
231  break;
232  }
233  }
234  } else {
235  for (std::size_t j = 0; j < localSize(); j++)
236  {
237  if (std::norm(state[j]) > tolerance)
238  {
239  bit_is_one = 1;
240  break;
241  }
242  }
243  }
244  }
245 
246  done:
247  int glb_bit_is_zero, glb_bit_is_one;
248 #ifdef INTELQS_HAS_MPI
249  MPI_Allreduce(&bit_is_zero, &glb_bit_is_zero, 1, MPI_INT, MPI_LOR, MPI_COMM_WORLD);
250  MPI_Allreduce(&bit_is_one , &glb_bit_is_one , 1, MPI_INT, MPI_LOR, MPI_COMM_WORLD);
251 #else
252  glb_bit_is_zero = bit_is_zero;
253  glb_bit_is_one = bit_is_one ;
254 #endif
255 
256  if (glb_bit_is_zero == 1 && glb_bit_is_one == 0)
257  return false;
258  else if (glb_bit_is_zero == 0 && glb_bit_is_one == 1)
259  return true;
260  else
261  assert(false); // this should never be called
262 
263  return false; // dummy return
264 }
