211 template <class Type>
212 void NoisyQureg<Type>::NoiseGate(unsigned const qubit )
213 {
214  BaseType t = TimeFromLastGate[qubit] ;
215  if (t==0) return;
216 
217  BaseType p_X , p_Y , p_Z ;
218  p_X = (1. - std::exp(-t/T_1) )/4.;
219  p_Y = (1. - std::exp(-t/T_1) )/4.;
220  p_Z = (1. - std::exp(-t/T_2) )/2. + (1. - std::exp(-t/T_1) )/4.;
221  assert( p_X>0 && p_Y>0 && p_Z>0 );
222 
223  // Computation of the standard deviations for the noise gate parameters
224  BaseType s_X , s_Y , s_Z ;
225  s_X = std::sqrt( -std::log(1.-p_X) );
226  s_Y = std::sqrt( -std::log(1.-p_Y) );
227  s_Z = std::sqrt( -std::log(1.-p_Z) );
228 
229  // Generate angle and direction of Pauli rotation for Pauli-twirld noise channel
230  BaseType v_X , v_Y , v_Z;
231  v_X = gaussian_RNG(generator) * s_X /2.;
232  v_Y = gaussian_RNG(generator) * s_Y /2.;
233  v_Z = gaussian_RNG(generator) * s_Z /2.;
234 
235  // Compose the 3-dimensional rotation: R_X(v_X).R_Y(v_Y).R_Z(v_Z)
236  // | cos Y cos Z -cos Y sin Z sin Y |
237  // R = | sin X sin Y cos Z + cos X sin Z -sin X sin Y sin Z + cos X cos Z -sin X cos Y |
238  // | -cos X sin Y cos Z + sin X sin Z cos X sin Y sin Z + sin X cos Z cos X cos Y |
239  //
240  // | cos X sin Y sin Z + sin X cos Z + sin X cos Y |
241  // u = | sin Y + cos X sin Y cos Z - sin X sin Z |
242  // | sin X sin Y cos Z + cos X sin Z + cos Y sin Z |
243  std::vector<BaseType> u = { std::cos(v_X) * std::sin(v_Y) * std::sin(v_Z) +
244  std::sin(v_X) * std::cos(v_Z) +
245  std::sin(v_X) * std::cos(v_Y),
246  std::sin(v_Y) +
247  std::cos(v_X) * std::sin(v_Y) * std::cos(v_Z) -
248  std::sin(v_X) * std::sin(v_Z),
249  std::sin(v_X) * std::sin(v_Y) * std::cos(v_Z) +
250  std::cos(v_X) * std::sin(v_Z) +
251  std::cos(v_Y) * std::sin(v_Z) };
252  BaseType norm_u = std::sqrt( std::norm(u[0]) + std::norm(u[1]) + std::norm(u[2]) );
253  std::vector<BaseType> axis = { u[0]/norm_u , u[1]/norm_u , u[2]/norm_u };
254  // Compute the angle of rotation
255  BaseType trace_R = std::cos(v_Y) * std::cos(v_Z)
256  - std::sin(v_X) * std::sin(v_Y) * std::sin(v_Z)
257  + std::cos(v_X) * std::cos(v_Z)
258  + std::cos(v_X) * std::cos(v_Y) ;
259  BaseType angle = std::acos( (trace_R-1.)/2. );
260  if (false) std::cout << " angle of rotation = " << angle << "\n";
261  // Alternative expression for the angle, from Wikipedia.
262  if (false) std::cout << " angle of rotation (alternative formula) = " << std::asin( norm_u/2. ) << "\n";
263 
264 
265  // Costruct the 1/2-spin matrix corresponding to the axis above
266  // sigma_axis
267  // and use it to implement the single-qubit rotation :
268  // rot = exp( i * angle * sigma_axis )
269  openqu::TinyMatrix<Type, 2, 2, 32> rot;
270  BaseType s(std::sin(angle/2.)) , c(std::cos(angle/2.)) ;
271  rot(0, 0) = Type( c , s*axis[2] );
272  rot(0, 1) = Type( s*axis[1] , s*axis[0] );
273  rot(0, 1) = Type( -s*axis[1] , s*axis[0] );
274  rot(1, 1) = Type( c , -s*axis[2] );
275 
276  // Apply the noise gate
277  QbitRegister<Type>::apply1QubitGate(qubit,rot);
278 }
