198 double QubitRegister<Type>::Entropy()
199 {
200 
201  std::size_t lcl = localSize();
202  double local_Hp = 0;
203 
204  if(timer) timer->Start("ENT", 0);
205 
206  double ttot = 0., ttmp1 = sec();
207 #if defined(__ICC) || defined(__INTEL_COMPILER)
208 #pragma omp parallel for reduction(+ : local_Hp)
209 #else
210  TODO(Remember to find 'omp parallel for simd reduction' equivalent for gcc)
211 #endif
212  for (std::size_t i = 0; i < lcl; i++)
213  {
214  double pj = std::norm(state[i]) ;
215  if (pj != double(0.))
216  {
217  local_Hp -= pj * std::log(pj);
218  }
219  }
220 
221  double global_Hp;
222 #ifdef INTELQS_HAS_MPI
223  MPI_Comm comm = openqu::mpi::Environment::comm();
224  // MPI_Allreduce(&local_Hp, &global_Hp, 1, MPI_DOUBLE, MPI_SUM, comm);

226 #else
227  global_Hp = local_Hp;
228 #endif
229 
230  ttot = sec() - ttmp1;
231 
232  if (timer)
233  {
234  double datab = D(sizeof(state[0])) * D(lcl) / ttot;
235  timer->record_sn(ttot, datab / ttot);
236  timer->Stop();
237  }
238 
239  return global_Hp / (double)log(double(2.0));

245 template <class Type>
246 std::vector<double> QubitRegister<Type>::GoogleStats()
247 {
248  std::vector <double> stats;
249 
250  std::size_t lcl = localSize();
251  double two2n = D(globalSize());
252 
253  double entropy = 0, avgselfinfo=0,
254  m2 = 0, m3 = 0, m4 = 0, m5 = 0, m6 = 0,
255  m7 = 0, m8 = 0, m9 = 0, m10 = 0;
256 
257  if(timer) timer->Start("ENT", 0);
258 
259  double ttot = 0., ttmp1 = sec();
260 
261 #if defined(__ICC) || defined(__INTEL_COMPILER)
262 #pragma omp parallel for reduction(+ : entropy, avgselfinfo, m2, m3, m4, m5, m6, m7, m8, m9, m10)
263 #else
264  TODO(Remember to find 'omp parallel for simd reduction' equivalent for gcc)
265 #endif
266  #pragma simd reduction(+ : entropy, avgselfinfo, m2, m3, m4, m5, m6, m7, m8, m9, m10)
267  // #pragma novector
268  for (std::size_t i = 0; i < lcl; i++)
269  {
270  double pj = std::norm(state[i]) ;
271  if (pj != double(0.))
272  {
273  double nl = log(pj);
274  // double nl = pj*pj;
275  entropy -= pj * nl;
276  avgselfinfo -= nl;
277  }
278  double pj2 = pj * pj,
279  pj3 = pj2 * pj,
280  pj4 = pj2 * pj2,
281  pj5 = pj3 * pj2,
282  pj6 = pj3 * pj3,
283  pj7 = pj4 * pj3,
284  pj8 = pj4 * pj4,
285  pj9 = pj5 * pj4,
286  pj10 = pj5 * pj5;
287  m2 += pj2;
288  m3 += pj3;
289  m4 += pj4;
290  m5 += pj5;
291  m6 += pj6;
292  m7 += pj7;
293  m8 += pj8;
294  m9 += pj9;
295  m10 += pj10;
296  }
297 
298  double global_entropy;
299  double global_avgselfinfo;
300 #ifdef INTELQS_HAS_MPI
301  MPI_Comm comm = openqu::mpi::Environment::comm();
302  MPI_Allreduce_x(&entropy, &global_entropy, MPI_SUM, comm);
303  MPI_Allreduce_x(&avgselfinfo, &global_avgselfinfo, MPI_SUM, comm);
304 #else
305  global_entropy = entropy;
306  global_avgselfinfo = avgselfinfo;
307 #endif
308  global_entropy /= (double)std::log(double(2.0));
309  stats.push_back(global_entropy);
310  global_avgselfinfo /= (double)log(double(2.0));
311  global_avgselfinfo /= two2n;
312  stats.push_back(global_avgselfinfo);
313 
314  // compute moments
315  std::vector <double> m = {m2, m3, m4, m5, m6, m7, m8, m9, m10},
316  factor(m.size()),
317  global_m(m.size());
318  double factorial = 1.0;
319  for(auto i = 0; i < m.size(); i++)
320  {
321  auto k = i + 2;
322  factorial *= D(k);
323  factor[i] = pow(two2n, D(k - 1)) / factorial;
324 
325  m[i] *= factor[i];
326 #ifdef INTELQS_HAS_MPI
327  MPI_Allreduce_x(&(m[i]), &(global_m[i]), MPI_SUM, comm);
328 #else
329  global_m[i] = m[i];
330 #endif
331  stats.push_back(global_m[i]);
332  }
333 
334  ttot = sec() - ttmp1;
335 
336  if (timer)
337  {
338  double datab = D(sizeof(state[0])) * D(lcl) / ttot;
339  timer->record_sn(ttot, datab / ttot);
340  timer->Stop();
341  }
342 
343  return stats;
344 }

357  BaseType &cumulative_probability, Permutation *permutation)

366  std::real(state[i]), std::imag(state[i]),
367  bin.c_str(), std::norm(state[i]) );
368  str = str + s;
369  cumulative_probability += std::norm(state[i]);
370  }
371  return std::string(str);
