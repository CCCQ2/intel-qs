154 template <class Type>
155 bool QbitRegister<Type>::getClassicalValue(unsigned qubit, BaseType tolerance) const
156 {
157  int myrank = openqu::mpi::Environment::rank();
158  int nprocs = openqu::mpi::Environment::size();
159  unsigned log2_nprocs = openqu::ilog2(openqu::mpi::Environment::size());
160  std::size_t delta = 1UL << qubit;
161  std::size_t M = nqbits - log2_nprocs;
162 
163  int bit_is_zero = 0, bit_is_one = 0;
164  if (qubit < M) {
165  for (std::size_t i = 0; i < localSize(); i += 2 * delta) {
166  for (std::size_t j = 0; j < delta; ++j) {
167  if (std::norm(state[i + j]) > tolerance) {bit_is_zero = 1; goto done;}
168  if (std::norm(state[i + j + delta]) > tolerance) {bit_is_one = 1; goto done;}
169  }
170  }
171  } else {
172  std::size_t src_glb_start = UL(myrank) * localSize();
173  if (check_bit(src_glb_start, qubit) == 0) {
174  for (std::size_t j = 0; j < localSize(); j++)
175  if (std::norm(state[j]) > tolerance) {bit_is_zero = 1; break;}
176  } else {
177  for (std::size_t j = 0; j < localSize(); j++)
178  if (std::norm(state[j]) > tolerance) {bit_is_one = 1; break;}
179  }
180  }
181 
182  done:
183  int glb_bit_is_zero, glb_bit_is_one;
184  MPI_Allreduce(&bit_is_zero, &glb_bit_is_zero, 1, MPI_INT, MPI_LOR, MPI_COMM_WORLD);
185  MPI_Allreduce(&bit_is_one, &glb_bit_is_one, 1, MPI_INT, MPI_LOR, MPI_COMM_WORLD);
186  if (glb_bit_is_zero == 1 && glb_bit_is_one == 0) return false;
187  else if (glb_bit_is_zero == 0 && glb_bit_is_one == 1) return true;
188  else assert(false); // this should never be called
189  return false; // dummy return
190 }
