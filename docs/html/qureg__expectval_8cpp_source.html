463 // observable: 1==PauliX , 2==PauliY , 3==PauliZ
464 template <class Type>
465 void QbitRegister<Type>::expectationValue(std::vector<unsigned> &qubits, std::vector<unsigned> &observables, BaseType &sum, BaseType coeff)
466 {
467  assert( qubits.size() == observables.size() );
468 
469 // G is matrix from change of basis Y --> Z, such that Ginv.Z.G = Y
470  openqu::TinyMatrix<Type, 2, 2, 32> G;
471  BaseType f = 1. / std::sqrt(2.);
472  G(0, 0) = G(1, 0) = Type(f , 0.);
473  G(0, 1) = Type(0.,-f);
474  G(1, 1) = Type(0., f);
475 // G^dagger = G^-1
476  openqu::TinyMatrix<Type, 2, 2, 32> Ginv;
477  Ginv(0, 0) = Ginv(0, 1) = Type(f , 0.);
478  Ginv(1, 0) = Type(0., f);
479  Ginv(1, 1) = Type(0.,-f);
480 
481  for (std::size_t i=0; i<qubits.size(); i++)
482  {
483  if (observables[i]==1)
484  applyHadamard(qubits[i]);
485  else if (observables[i]==2)
486  apply1QubitGate(qubits[i],G);
487  else if (observables[i]==3)
488  continue;
489  else
490  assert(0); // should never be called
491  }
492 
493 // compute the expectation value
494  MPI_Comm comm = openqu::mpi::Environment::comm();
495  std::size_t myrank = openqu::mpi::Environment::rank();
496  BaseType local_value = 0;
497  std::size_t glb_start = UL(myrank) * localSize();
498 // integer in binary notation with 1 located at the position of the qubits
499  std::size_t y=0;
500  for (std::size_t i=0; i<qubits.size(); i++)
501  y += 1 << qubits[i];
502 
503 #pragma omp parallel
504 {
505  std::size_t x;
506  #pragma omp for reduction(+ : local_value)
507  for(std::size_t i = 0; i < localSize(); i++)
508  {
509  x = glb_start + i;
510  if ( Hamming_weight( x & y ) & 1 ) // odd number of 1 in the qubits involved
511  local_value -= std::abs(state[i]) * std::abs(state[i]);
512  else
513  local_value += std::abs(state[i]) * std::abs(state[i]);
514  }
515 }
516  BaseType global_value;
517  // MPI_Allreduce(&local_value, &global_value, 1, MPI_DOUBLE, MPI_SUM, comm);
518  MPI_Allreduce_x(&local_value, &global_value, MPI_SUM, comm);
519 
520 // update sum
521  sum += coeff * global_value;
522 
523 // undo the quantum gates to recover the initial state |psi>
524  for (std::size_t i=0; i<qubits.size(); i++)
525  {
526  if (observables[i]==1)
527  applyHadamard(qubits[i]);
528  else if (observables[i]==2)
529  apply1QubitGate(qubits[i],Ginv);
530  else if (observables[i]==3)
531  continue;
532  else
533  assert(0); // should never be called
534  }
535 }
