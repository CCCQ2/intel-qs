218 template <class Type>

220 {
221  unsigned num_qubits = this->num_qubits;
222  // Implement the noise gate
223  NoiseGate(qubit);
224  // Increase the idle time for all the qubits (TODO no gate parallelism is assumed)
225  for (unsigned q = 0; q<num_qubits; q++)
226  time_from_last_gate[q] += time_one_qubit_experimental_gate ;
227  // Reset the time elapsed from last logical gate on the specific qubit
228  time_from_last_gate[qubit]=0.;
229  // Update counter for (logical) one-qubit gates
230  one_qubit_experimental_gate_count++;
231  experimental_gate_count_matrix[qubit*num_qubits+qubit]++;
232 }

260 template <class Type>
261 void NoisyQureg<Type>::NoiseGate(unsigned const qubit )
262 {
263  BaseType t = time_from_last_gate[qubit] ;
264  if (t==0) return;
265 
266  BaseType p_X , p_Y , p_Z ;
267  p_X = (1. - std::exp(-t/T_1) )/4.;
268  p_Y = (1. - std::exp(-t/T_1) )/4.;
269  p_Z = (1. - std::exp(-t/T_2) )/2. + (1. - std::exp(-t/T_1) )/4.;
270  assert( p_X>0 && p_Y>0 && p_Z>0 );
271 
272  // Computation of the standard deviations for the noise gate parameters
273  BaseType s_X , s_Y , s_Z ;
274  s_X = std::sqrt( -std::log(1.-p_X) );
275  s_Y = std::sqrt( -std::log(1.-p_Y) );
276  s_Z = std::sqrt( -std::log(1.-p_Z) );
277 
278  // Generate angle and direction of Pauli rotation for Pauli-twirld noise channel
279  BaseType v_X , v_Y , v_Z;
280  v_X = gaussian_RNG(generator) * s_X /2.;
281  v_Y = gaussian_RNG(generator) * s_Y /2.;
282  v_Z = gaussian_RNG(generator) * s_Z /2.;
283 
284  // Direct construction of the 2x2 matrix corresponding to the noise gate
285  // U_noise = exp(-i v_X X) * exp(-i v_Y Y) * exp(-i v_Z Z)
286  // Helpful quantities:
287  // (A) = cos v_z -i sin v_z
288  // (B) = cos v_x * cos v_Y -i sin v_X * sin v_Y
289  // (C) = cos v_x * sin v_Y -i sin v_X * cos v_Y
290  // Then :
291  // | A*B -A'*C' |
292  // U_noise = | |
293  // | A*C A'*B' |
294 
295  Type A , B , C ;
296  A = { std::cos(v_Z) , -std::sin(v_Z) };
297  B = { std::cos(v_X)*std::cos(v_Y) , -std::sin(v_X)*std::sin(v_Y) };
298  C = { std::cos(v_X)*std::sin(v_Y) , -std::sin(v_X)*std::cos(v_Y) };
299 
300  openqu::TinyMatrix<Type, 2, 2, 32> U_noise;
301  U_noise(0, 0) = A*B;
302  U_noise(0, 1) = -std::conj(A)*std::conj(C);
303  U_noise(1, 0) = A*C;
304  U_noise(1, 1) = std::conj(A)*std::conj(B);
305 
306  // Apply the noise gate
307  QubitRegister<Type>::Apply1QubitGate(qubit,U_noise);
308 }

322 void NoisyQureg<Type>::NoiseGate_OLD(unsigned const qubit )
323 {
324  BaseType t = time_from_last_gate[qubit] ;
325  if (t==0) return;
326 
327  BaseType p_X , p_Y , p_Z ;
328  p_X = (1. - std::exp(-t/T_1) )/4.;
329  p_Y = (1. - std::exp(-t/T_1) )/4.;
330  p_Z = (1. - std::exp(-t/T_2) )/2. + (1. - std::exp(-t/T_1) )/4.;
331  assert( p_X>0 && p_Y>0 && p_Z>0 );
332 
333  // Computation of the standard deviations for the noise gate parameters
334  BaseType s_X , s_Y , s_Z ;
335  s_X = std::sqrt( -std::log(1.-p_X) );
336  s_Y = std::sqrt( -std::log(1.-p_Y) );
337  s_Z = std::sqrt( -std::log(1.-p_Z) );
338 
339  // Generate angle and direction of Pauli rotation for Pauli-twirld noise channel
340  BaseType v_X , v_Y , v_Z;
341  v_X = gaussian_RNG(generator) * s_X /2.;
342  v_Y = gaussian_RNG(generator) * s_Y /2.;
343  v_Z = gaussian_RNG(generator) * s_Z /2.;
344 
345  // Compose the 3-dimensional rotation: R_X(v_X).R_Y(v_Y).R_Z(v_Z)
346  // | cos Y cos Z -cos Y sin Z sin Y |
347  // R = | sin X sin Y cos Z + cos X sin Z -sin X sin Y sin Z + cos X cos Z -sin X cos Y |
348  // | -cos X sin Y cos Z + sin X sin Z cos X sin Y sin Z + sin X cos Z cos X cos Y |
349  //
350  // | cos X sin Y sin Z + sin X cos Z + sin X cos Y |
351  // u = | sin Y + cos X sin Y cos Z - sin X sin Z |
352  // | sin X sin Y cos Z + cos X sin Z + cos Y sin Z |
353 
354  std::vector<BaseType> R = { std::cos(v_Y)*std::cos(v_Z) ,
355  -std::cos(v_Y)*std::sin(v_Z) ,
356  std::sin(v_Y) ,
357  std::sin(v_X)*std::sin(v_Y)*std::cos(v_Z) + std::cos(v_X)*std::sin(v_Z) ,
358  -std::sin(v_X)*std::sin(v_Y)*std::sin(v_Z) + std::cos(v_X)*std::cos(v_Z) ,
359  -std::sin(v_X) * std::cos(v_Y),
360  -std::cos(v_X)*std::sin(v_Y)*std::cos(v_Z) + std::sin(v_X)*std::sin(v_Z),
361  std::cos(v_X)*std::sin(v_Y)*std::sin(v_Z) + std::sin(v_X)*std::cos(v_Z),
362  std::cos(v_X)*std::cos(v_Y) };
363 
364  std::vector<BaseType> u = { R[3*2+1] - R[3*1+2] ,
365  R[3*0+2] - R[3*2+0] ,
366  R[3*1+0] - R[3*0+1] } ;
367  BaseType norm_u = std::sqrt( std::norm(u[0]) + std::norm(u[1]) + std::norm(u[2]) );
368  std::vector<BaseType> axis = { u[0]/norm_u , u[1]/norm_u , u[2]/norm_u };
369  // Compute the angle of rotation
370  BaseType trace_R = R[0] + R[4] + R[8] ;
371  BaseType angle = std::acos( (trace_R-1.)/2. );
372 
373  // Alternative expression for the angle, from Wikipedia:
374  BaseType angle_alternative = std::asin( norm_u/2. ) ;
375  // if 0<angle<Pi/2 then angle = angle_alternative
376  // if Pi/2<angle<Pi then angle = Pi - angle_alternative
377 
378  // It may happen that "axis" still requires a "minus sign"
379  if (false)
380  {
381  std::vector<double> R_u , R_minus_u ;
382  BaseType si(std::sin(angle)) , co(std::cos(angle)) ;
383  R_u = { co + axis[0]*axis[0]*(1.-co) ,
384  axis[0]*axis[1]*(1.-co) - axis[2]*si ,
385  axis[0]*axis[2]*(1.-co) + axis[1]*si ,
386  axis[0]*axis[1]*(1.-co) + axis[2]*si ,
387  co + axis[1]*axis[1]*(1.-co) ,
388  axis[1]*axis[2]*(1.-co) - axis[0]*si ,
389  axis[0]*axis[2]*(1.-co) - axis[1]*si ,
390  axis[1]*axis[2]*(1.-co) + axis[0]*si ,
391  co + axis[2]*axis[2]*(1.-co) };
392  R_minus_u = { co + axis[0]*axis[0]*(1.-co) ,
393  axis[0]*axis[1]*(1.-co) + axis[2]*si ,
394  axis[0]*axis[2]*(1.-co) - axis[1]*si ,
395  axis[0]*axis[1]*(1.-co) - axis[2]*si ,
396  co + axis[1]*axis[1]*(1.-co) ,
397  axis[1]*axis[2]*(1.-co) + axis[0]*si ,
398  axis[0]*axis[2]*(1.-co) + axis[1]*si ,
399  axis[1]*axis[2]*(1.-co) - axis[0]*si ,
400  co + axis[2]*axis[2]*(1.-co) };
401 
402 // print_vector(R,"rotation matrix:\n");
403 // print_vector(R_u,"reconstructed with u:\n");
404 // print_vector(R_minus_u,"reconstructed with -u:\n");
405 
406  // u was defined up to a sign. To resolve the ambiguity:
407  BaseType R_element , Ru_element;
408  if (axis[0] != 0.)
409  {
410  R_element = std::cos(v_X) * std::sin(v_Y) * std::sin(v_Z) + std::sin(v_X) * std::cos(v_Z) ;
411  Ru_element = axis[2]*axis[1]*(1-std::cos(angle)) + axis[0]*std::sin(angle) ;
412  if ( std::abs(R_element - Ru_element)> 1e-7 ) std::cout << " wrong sign from axis[0]\n";
413 // else std::cout << " correct sign from axis[0]\n";
414  }
415  if (axis[1] != 0.)
416  {
417  R_element = std::sin(v_Y) ;
418  Ru_element = axis[2]*axis[0]*(1-std::cos(angle)) + axis[1]*std::sin(angle) ;
419  if ( std::abs(R_element - Ru_element)> 1e-7 ) std::cout << " wrong sign from axis[1]\n";
420 // else std::cout << " correct sign from axis[1]\n";
421  }
422  if (axis[2] != 0.)
423  {
424  R_element = std::sin(v_X) * std::sin(v_Y) * std::cos(v_Z) + std::cos(v_X) * std::sin(v_Z) ;
425  Ru_element = axis[0]*axis[1]*(1-std::cos(angle)) + axis[2]*std::sin(angle) ;
426  if ( std::abs(R_element - Ru_element)> 1e-7 ) std::cout << " wrong sign from axis[2]\n";
427 // else std::cout << " correct sign from axis[2]\n";
428  }
429  }
430 
431  // Compute the angle of rotation
432  trace_R = std::cos(v_Y) * std::cos(v_Z)
433  - std::sin(v_X) * std::sin(v_Y) * std::sin(v_Z)
434  + std::cos(v_X) * std::cos(v_Z)
435  + std::cos(v_X) * std::cos(v_Y) ;
436  angle = std::acos( (trace_R-1.)/2. );
437  if (false) std::cout << " angle of rotation = " << angle << "\n";
438  // Alternative expression for the angle, from Wikipedia.
439  if (false) std::cout << " angle of rotation (alternative formula) = " << std::asin( norm_u/2. ) << "\n";
440 
441 
442  // Costruct the 1/2-spin matrix corresponding to the axis above
443  // sigma_axis
444  // and use it to implement the single-qubit rotation :
445  // rot = exp( i * angle * sigma_axis )
446  openqu::TinyMatrix<Type, 2, 2, 32> rot;
447  BaseType s(std::sin(angle/2.)) , c(std::cos(angle/2.)) ;
448  rot(0, 0) = Type( c , s*axis[2] );
449  rot(0, 1) = Type( s*axis[1] , s*axis[0] );
450  rot(1, 0) = Type( -s*axis[1] , s*axis[0] );
451  rot(1, 1) = Type( c , -s*axis[2] );
452
