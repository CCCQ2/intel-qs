376  std::size_t chunk = localSize() / num_threads;
377  std::size_t beginning = thread_id * chunk, end = (thread_id + 1) * chunk;

380  int errcode = vslNewStream(&stream, VSL_BRNG_MCG31, base_index);

383  vslSkipAheadStream(stream, num_skip);

385  (double *)&state[beginning], 0.0, 1.0);
