33 
34  /*
35  Steps: 1. 2. 3. 4.
36  i j | i j | i j | i j
37  s1 d1 | s1 d1 | s1 d1&s1 | s1&d1 d1&s1
38  s2 d2 | s2 d2 | s2&d2 d2 | s2&d2 d2&s2
39  T T | d2 s1 | d2&s2 s1&d1 | T T
40 
41  */
42 
43  int tag1 = 1, tag2 = 2;
44  std::size_t glb_start = UL(myrank) * localSize();
45  unsigned int itask, jtask;
46 
47  if (check_bit(glb_start, T) == 0) {
48  itask = myrank;
49  jtask = itask + (1 << (T - M));
50  } else {
51  jtask = myrank;
52  itask = jtask - (1 << (T - M));
53  }
54 
55  // 1. allocate temp buffer
56  Type *tmpstate = tmpspace();
57  std::size_t lcl_size_half = localSize() / 2L;
58  assert(lcl_size_half <= std::numeric_limits<int>::max());
59 
60 
61 #if 0
62  size_t lcl_chunk = 128;
63  if (lcl_chunk > lcl_size_half)
64  lcl_chunk = lcl_size_half;
65  else
66  assert((lcl_size_half % lcl_chunk) == 0);
67 #else
68  size_t lcl_chunk = tmpSize();
69  if (lcl_chunk != lcl_size_half) {
70  fprintf(stderr, "Need to fix chunking first\n");
71  assert(0);
72  }
73 #endif
74 
75  double t, tnet = 0;
76  for(size_t c = 0; c < lcl_size_half; c += lcl_chunk) {
77  if (itask == myrank) { // this is itask
78  // 2. src sends s1 to dst into dT
79  // dst sends d2 to src into dT
80  t = sec();
81  MPI_Sendrecv_x(&(state[c]), lcl_chunk, jtask, tag1, &(tmpstate[0]),
82  lcl_chunk, jtask, tag2, comm, &status);
83  tnet += sec() - t;
84 
85  // 3. src and dst compute
86  if (M - C == 1) {
87  Loop_SN(0L, lcl_chunk, state, tmpstate, lcl_size_half, 0L, m, specialize, timer);
88  } else {
89  Loop_DN((UL(1) << C), lcl_size_half, C, state, tmpstate, lcl_size_half, 0L,
90  m, specialize, timer);
91  }
92 
93  t = sec();
94  MPI_Sendrecv_x(&(tmpstate[0]), lcl_chunk, jtask, tag1, &(state[c]),
95  lcl_chunk, jtask, tag2, comm, &status);
96  tnet += sec() - t;
97 
98  } else { // this is jtask
99  // 2. src sends s1 to dst into dT
100  // dst sends d2 to src into dT
101  t = sec();
102  MPI_Sendrecv_x(&(state[lcl_size_half + c]), lcl_chunk, itask, tag2,
103  &(tmpstate[0]), lcl_chunk, itask, tag1, comm,
104  &status);
105  tnet += sec() - t;
106 
107  if (M - C == 1) {
108  // this is intentional special case: nothing happens
109  } else {

115  &(state[lcl_size_half + c]), lcl_chunk, itask, tag1,
116  comm, &status);
117  tnet += sec() - t;
118  }
119  }
120

126 #endif
