33 double QubitRegister<Type>::HP_Distrpair(unsigned control_position, unsigned target_position,
34  TM2x2<Type> const&m)
35 {
36 #ifdef INTELQS_HAS_MPI
37  MPI_Status status;
38  MPI_Comm comm = openqu::mpi::Environment::comm();
39  std::size_t myrank = openqu::mpi::Environment::rank();
40 
41  assert(target_position < num_qubits);
42  assert(control_position < num_qubits);
43  std::size_t M = num_qubits - openqu::ilog2(openqu::mpi::Environment::size());
44  std::size_t C = UL(control_position), T = UL(target_position);
45 
46  // Steps: 1. 2. 3. 4.
47  // i j | i j | i j | i j
48  // s1 d1 | s1 d1 | s1 d1&s1 | s1&d1 d1&s1
49  // s2 d2 | s2 d2 | s2&d2 d2 | s2&d2 d2&s2
50  // T T | d2 s1 | d2&s2 s1&d1 | T T
51 
52  int tag1 = 1, tag2 = 2;
53  std::size_t glb_start = UL(myrank) * localSize();
54  unsigned int itask, jtask;
55 
56  if (check_bit(glb_start, T) == 0)
57  {
58  itask = myrank;
59  jtask = itask + (1 << (T - M));
60  }
61  else
62  {
63  jtask = myrank;
64  itask = jtask - (1 << (T - M));
65  }
66 
67  // 1. allocate temp buffer
68  Type *tmp_state = TmpSpace();
69  std::size_t lcl_size_half = localSize() / 2L;
70  assert(lcl_size_half <= std::numeric_limits<int>::max());
71 
72 
73 #if 0
74  size_t lcl_chunk = 128;
75  if (lcl_chunk > lcl_size_half)
76  lcl_chunk = lcl_size_half;
77  else
78  assert((lcl_size_half % lcl_chunk) == 0);
79 #else
80  size_t lcl_chunk = TmpSize();
81  if (lcl_chunk != lcl_size_half)
82  {
83  fprintf(stderr, "Need to fix chunking first\n");
84  assert(0);
85  }
86 #endif
87 
88  double t, tnet = 0;
89  for(size_t c = 0; c < lcl_size_half; c += lcl_chunk)
90  {
91  if (itask == myrank) // this is itask
92  {
93  // 2. src sends s1 to dst into dT
94  // dst sends d2 to src into dT
95  t = sec();
96  MPI_Sendrecv_x(&(state[c]), lcl_chunk, jtask, tag1, &(tmp_state[0]),
97  lcl_chunk, jtask, tag2, comm, &status);
98  tnet += sec() - t;
99 
100  // 3. src and dst compute
101  if (M - C == 1) {
102  Loop_SN(0L, lcl_chunk, state, tmp_state, lcl_size_half, 0L, m, specialize, timer);
103  } else {
104  Loop_DN((UL(1) << C), lcl_size_half, C, state, tmp_state, lcl_size_half, 0L,
105  m, specialize, timer);
106  }
107 
108  t = sec();
109  MPI_Sendrecv_x(&(tmp_state[0]), lcl_chunk, jtask, tag1, &(state[c]),
110  lcl_chunk, jtask, tag2, comm, &status);
111  tnet += sec() - t;
112  }
113  else // this is jtask
114  {
115  // 2. src sends s1 to dst into dT
116  // dst sends d2 to src into dT
117  t = sec();
118  MPI_Sendrecv_x(&(state[lcl_size_half + c]), lcl_chunk, itask, tag2,
119  &(tmp_state[0]), lcl_chunk, itask, tag1, comm,
120  &status);
121  tnet += sec() - t;
122 
123  if (M - C == 1)
124  {} // this is intentional special case: nothing happens
125  else
126  Loop_DN((UL(1) << C), lcl_size_half, C, tmp_state, state, 0L, 0L, m, specialize, timer);
127 
128  t = sec();
129  MPI_Sendrecv_x(&(tmp_state[0]), lcl_chunk, itask, tag2,
130  &(state[lcl_size_half + c]), lcl_chunk, itask, tag1,
131  comm, &status);
132  tnet += sec() - t;
133  }
134  }
135 
136  double netsize = 2.0 * sizeof(Type) * 2.0 * D(lcl_size_half), netbw = netsize / tnet;
137  if (timer) timer->record_cm(tnet, netbw);
138 
139 #else
140  assert(0);
141 #endif
142 
143  return 0.0;
144 }

181  std::size_t src_glb_start = 0;
182  // check for special case of diagonal
183  bool diagonal = (m[0][1].real() == 0. && m[0][1].imag() == 0. &&
184  m[1][0].real() == 0. && m[1][0].imag() == 0.);

190  #if 0

193  if (m[0][1].real() == 0. && m[0][1].imag() == 0. &&
194  m[1][0].real() == 0. && m[1][0].imag() == 0.)

199  md(0, 0) = Type(1., 0.);

202  md(3, 3) = m[1][1];
203 
204  ApplyDiag(control, qubit, md);
205  assert(eind - sind == localSize());
206  assert(fusion == false);
207 
208  return true;
209  }
210  else
211  #endif
212  {
213  std::size_t cpstrideexp = C - M;
214 
215  if(C < M && T < M)
216  {
217  if(C > T)
218  {
219  // special case when we are blocking in LLC
220  // case when C stride is bigger than LLC block size
221  // in this case, we only update state if we are
222  // within part of the vector that has Cth bit set to
223  // one, since this is control gate
224  // Otherwise, we skip computation all together
225  if((C >= log2llc) && (localSize() > (eind - sind)))
226  {
227  if(check_bit(sind, C) == 1)
228  {
229  Loop_DN(sind, eind, T, state, state,
230  0, 1UL<<T, m, specialize, timer);
231  HasDoneWork = true;
232  }
233  }
234  else
235  {
236  Loop_TN(state,
237  sind, eind, 1UL<<C+1UL,
238  1UL<<C, 1UL<<C+1UL, 1UL<<T+1UL,
239  0L, 1UL<<T, 1UL<<T, m, specialize, timer);
240  HasDoneWork = true;
241  }
242 
243  }
244  else
245  {
246  Loop_TN(state,
247  sind, eind, 1UL<<T+1UL,
248  0L, 1UL<<T, 1UL<<C+1UL,
249  1UL<<C, 1UL<<C+1UL, 1UL<<T, m, specialize, timer);
250  HasDoneWork = true;
251  }
252  }
253  else if (C >= M && T < M)
254  {
255  assert(C > T);
256  if(((myrank >> cpstrideexp) % 2) != 0)
257  {
258  Loop_DN(sind, eind, T, state, state, 0L, (1UL << T), m, specialize, timer);
259  HasDoneWork = true;
260  }
261  }
262  else if (C >= M && T >= M)
263  {
264  if(((myrank >> cpstrideexp) % 2) != 0)
265  {
266  if (specialize && diagonal)
267  {
268  if (check_bit(src_glb_start, T) == 0 )
269  ScaleState(sind, eind, state, m[0][0], timer);
270  else
271  ScaleState(sind, eind, state, m[1][1], timer);
272  }
273  else
274  {
275  HP_Distrpair(T, m);
276  // printf("HPD 1\n");
277  }
278  HasDoneWork = true;
279  } else {
280  TODO(Way to fix problem with X and Y specializaion)
281  // openqu::mpi::Environment::remaprank(myrank);
282  }
283  }
284  else if (C < M && T >= M)
285  {
286  if (specialize && diagonal)
287  {
288  TM2x2<Type> md;
289  md[0][0] = {1.0, 0};
290  md[0][1] = md[1][0] = {0., 0.};
291  md[1][1] = (check_bit(src_glb_start, T) == 0) ? m[0][0] : m[1][1];
292  TODO(Insert Loop_SN specialization for this case)
293  Loop_DN(sind, eind, C, state, state, 0, 1UL<<C, md, specialize, timer);
294  }
295  else
296  {
297  HP_Distrpair(C, T, m);
298  // printf("HPD 2\n");
299  }
300  HasDoneWork = true;
301  }
302  else
303  assert(0);
304 
305  }
306  if(timer) timer->Stop();
307 
308  return HasDoneWork;
309 }
