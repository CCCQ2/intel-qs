168 bool QbitRegister<Type>::apply1QubitGate_helper(unsigned qubit_, TM2x2<Type> const&m,
169  std::size_t sind, std::size_t eind)
170 {
171  assert(qubit_ < nqbits);
172  unsigned qubit = (*permutation)[qubit_];
173  assert(qubit < nqbits);
174 
175  TODO(Add diagonal special case)
176 
177  unsigned myrank = openqu::mpi::Environment::rank();
178  unsigned nprocs = openqu::mpi::Environment::size();
179  unsigned log2_nprocs = openqu::ilog2(openqu::mpi::Environment::size());
180  unsigned M = nqbits - log2_nprocs;
181  std::size_t P = qubit;
182 
183  std::size_t src_glb_start = UL(myrank) * localSize();
184  // check for special case of diagonal
185  bool diagonal = (m[0][1].real() == 0. && m[0][1].imag() == 0. &&
186  m[1][0].real() == 0. && m[1][0].imag() == 0.);
187 
188  std::string gname = "SQG("+openqu::toString(P)+")::"+m.name;
189  if(timer) timer->Start(gname, P);
190 
191  if(P < M)
192  {
193  assert(eind - sind <= localSize());
194  Loop_DN(sind, eind, UL(P), state, state, 0UL, (1UL << P), m, specialize, timer);
195  }
196  else
197  {
198  assert(eind - sind == localSize());
199  if (specialize && diagonal) {
200  if (check_bit(src_glb_start, P) == 0 ) {
201  ScaleState(sind, eind, state, m[0][0], timer);
202  } else {
203  ScaleState(sind, eind, state, m[1][1], timer);
204  }
205  } else {
206  HP_Distrpair(P, m);
207  }
208  }
209  if(timer) timer->Stop();
210 
211 
212  return true;
213 }
