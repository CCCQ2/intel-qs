97 // recover the initial state |psi> by applying G^dagger

99  G(1, 0) = Type(0., f);
100  G(1, 1) = Type(0.,-f);

471 //------------------------------------------------------------------------------
472 // observable: 1==PauliX , 2==PauliY , 3==PauliZ
473 template <class Type>
474 void QubitRegister<Type>::ExpectationValue(std::vector<unsigned> &qubits,
475  std::vector<unsigned> &observables,
476  BaseType &sum, BaseType coeff)
477 {
478  assert( qubits.size() == observables.size() );
479 
480 // G is matrix from change of basis Y --> Z, such that Ginv.Z.G = Y
481  openqu::TinyMatrix<Type, 2, 2, 32> G;
482  BaseType f = 1. / std::sqrt(2.);
483  G(0, 0) = G(1, 0) = Type(f , 0.);
484  G(0, 1) = Type(0.,-f);
485  G(1, 1) = Type(0., f);
486 // G^dagger = G^-1
487  openqu::TinyMatrix<Type, 2, 2, 32> Ginv;
488  Ginv(0, 0) = Ginv(0, 1) = Type(f , 0.);
489  Ginv(1, 0) = Type(0., f);
490  Ginv(1, 1) = Type(0.,-f);
491 
492  for (std::size_t i=0; i<qubits.size(); i++)
493  {
494  if (observables[i]==1)
495  ApplyHadamard(qubits[i]);
496  else if (observables[i]==2)
497  Apply1QubitGate(qubits[i],G);
498  else if (observables[i]==3)
499  continue;
500  else
501  assert(0); // should never be called
502  }
503 
504 // compute the expectation value
505  std::size_t myrank = openqu::mpi::Environment::rank();
506  BaseType local_value = 0;
507  std::size_t glb_start = UL(myrank) * localSize();
508 // integer in binary notation with 1 located at the position of the qubits
509  std::size_t y=0;
510  for (std::size_t i=0; i<qubits.size(); i++)
511  {
512  y += 1 << qubits[i];
513  }
514 
515 #pragma omp parallel
516  {
517  std::size_t x;
518  #pragma omp for reduction(+ : local_value)
519  for(std::size_t i = 0; i < localSize(); i++)
520  {
521  x = glb_start + i;
522  if ( HammingWeight( x & y ) & 1 ) // odd number of 1 in the qubits involved
523  local_value -= std::abs(state[i]) * std::abs(state[i]);
524  else
525  local_value += std::abs(state[i]) * std::abs(state[i]);
526  }
527  }
528 
529  BaseType global_value ;
530 #ifdef INTELQS_HAS_MPI
531  MPI_Comm comm = openqu::mpi::Environment::comm();
532  // MPI_Allreduce(&local_value, &global_value, 1, MPI_DOUBLE, MPI_SUM, comm);
533  MPI_Allreduce_x(&local_value, &global_value, MPI_SUM, comm);
534 #else
535  global_value = local_value;
536 #endif
537 
538 // update sum
539  sum += coeff * global_value;
540 
541 // undo the quantum gates to recover the initial state |psi>
542  for (std::size_t i=0; i<qubits.size(); i++)
543  {
544  if (observables[i]==1)
545  ApplyHadamard(qubits[i]);
546  else if (observables[i]==2)
547  Apply1QubitGate(qubits[i],Ginv);
548  else if (observables[i]==3)
549  continue;
550  else
551  assert(0); // should never be called
552  }
553 }
